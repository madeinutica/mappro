Embed Instructions

To embed this map on your website, include the following code:

<!-- Include the Mapro embed script -->
<script src="https://your-domain.com/mapro-embed.js"></script>

<!-- Create a container for the map -->
<div id="my-map" style="height: 500px;"></div>

<!-- Initialize the map -->
<script>
  createMaproMap('my-map', {
    height: '500px',
    showHeader: true,
    clientId: 'your-client-id' // Optional: filter by specific client
  });
</script>
